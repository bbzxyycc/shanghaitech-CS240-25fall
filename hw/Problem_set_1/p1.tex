\problem{}

\par Arrange the following functions in order of increasing asymptotic growth rate.

\begin{itemize}
    \item \( f_1(n) = n^4 + \log n \)
    \item \( f_2(n) = n + \left[\log(n)\right]^4 \)
    \item \( f_3(n) = n \log n \)
\item \( f_4(n) = C_n^3 \)
\item \( f_5(n) = C_n^{n/2} \)
    \item \( f_6(n) = 2^n \)
    \item \( f_7(n) = n^{\log n} \)
\end{itemize}

\noindent \textbf{Hint:}  
Use Stirling’s approximation for factorials:  
\[
n! \sim \sqrt{2\pi n}\,\left(\tfrac{n}{e}\right)^n.
\]
\textbf{Solution}:

\noindent$f_1(n) = \Theta(n^4)$,

\noindent$f_2(n) = \Theta(n)$,

\noindent$f_3(n) = \Theta(nlogn)$,

\noindent$f_4(n) = \frac{n(n-1)(n-2)}{3!} = \Theta(n^{3})$,

\noindent$f_5(n) = \frac{n!}{(\frac{n}{2}!)^2} \sim \frac{\sqrt{2\pi n}(\frac{n}{e})^n}{(\sqrt{\pi n}(\frac{n}{2e})^n)^2} = \frac{2^n}{\sqrt{\frac{\pi n}{2}}}= \Theta(\frac{2^n}{\sqrt{n}})$,

\noindent$f_6(n) =  \Theta(2^n)$,

\noindent$f_7(n) =  (2^{\log n})^{\log n} = 2^{\log^2 n}= \Theta(2^{\log^2 n})$,


\noindent Thus:

$f_2(n)<f_3(n)<f_4(n)<f_1(n)<f_7(n)<f_5(n)<f_6(n)$
